'card-badges': function(t) { function pad(n) { return n < 10 ? '0' + n : '' + n; } function formatTime(secs) { var h = Math.floor(secs / 3600); var m = Math.floor((secs % 3600) / 60); var s = secs % 60; return (h > 0 ? h + ':' + pad(m) : pad(m)) + ':' + pad(s); } return t.get('card', 'shared', 'checklists', []).then(function(checklists) { var totalItems = 0; var checkedItems = 0; var totalTime = 0; (checklists || []).forEach(function(cl) { (cl.items || []).forEach(function(item) { totalItems++; if (item.checked) checkedItems++; totalTime += item.timerTotal || 0; if (item.timerStart) { totalTime += Math.floor((Date.now() - item.timerStart) / 1000); } (item.subitems || []).forEach(function(sub) { totalItems++; if (sub.checked) checkedItems++; totalTime += sub.timerTotal || 0; if (sub.timerStart) { totalTime += Math.floor((Date.now() - sub.timerStart) / 1000); } }); }); }); var badges = []; if (totalItems > 0) { badges.push({ text: '☑ ' + checkedItems + '/' + totalItems, color: checkedItems === totalItems ? 'green' : 'light-gray' }); } if (totalTime > 0) { badges.push({ text: '⏱ ' + formatTime(totalTime), color: 'light-gray', refresh: 10 }); } return badges; }); }
